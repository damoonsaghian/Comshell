; nano ~/.bash_profile
  export TZ="/usr/share/zoneinfo/<your_location>"
  sway
; mkdir -p ~/.config/sway
; nano ~/.config/sway/config
  default_floating_border pixel 1
  default_border pixel 1
  hide_edge_borders smart
  focus_follows_mouse no
  for_window [class="^(?!Emacs$)"] floating enable, \
    resize set 90 ppt 90 ppt, move position center, move scratchpad, scratchpad show
  for_window [class="^Emacs$" title=.] floating enable, \
    resize set 90 ppt 90 ppt, move position center
  for_window [class="^Emacs$" title="^projects$"] floating enable, \
    resize set 25 ppt 90 ppt, move position center
  exec --no-startup-id "emacs --name=''; swaymsg '[class=.] kill'; swaymsg exit"

or for "i3-wm":

; nano ~/.xinitrc
  exec i3
; mkdir -p ~/.config/i3
content of "~/.config/i3/config" is like the one written above for "sway", just:
  add this to the first line: # i3 config file (v4)
  add this line: exec compton --vsync opengl-swc
  in the last line, replace "swaymsg" with "i3-msg";

Firefox:
; sudo pacman -S firefox materia-gtk-theme
; mkdir -p ~/.config/gtk-3.0; nano ~/.config/gtk-3.0/settings.ini
  [Settings]
  gtk-theme-name = Materia-light-compact
  gtk-font-name = Sans
then add these to i3/sway config:
  for_window [class=Firefox window_type=normal] resize set 100 ppt 100 ppt, move position center
  bindsym Mod1+Tab exec --no-startup-id \
    "if [[ \\"$(i3-msg [workspace=__focused__ class=Firefox] scratchpad show)\\" =~ \\"false\\" ]]; \
    then if [[ \\"$(i3-msg [class=Firefox] scratchpad show)\\" =~ \\"false\\" ]]; \
         then firefox; fi fi"
